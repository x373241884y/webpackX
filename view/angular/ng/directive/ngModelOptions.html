ngModelOptions


描述：

设置模型怎么更新。使用ngmodeloptions你可以指定一个自定义，设定如何触发模型的更新。


指令优先级

优先级别0。


用法：

<ANY
        ng-model-options="">
...
</ANY>

属性：

参数名称 参数类型 描述
ngList string 分隔符

例子：

效果演示 
<!DOCTYPE html>
<html xmlns="http://www.w3.org/1999/xhtml" ng-app="optionsExample">
<head>
        <meta http-equiv="Content-Type" content="text/html; charset=utf-8" />
        <title>ngModelOptions例子</title>
        <script src="../angular-1.3.0.14/angular.js"></script>
        <script type="text/javascript">
        angular.module('optionsExample', [])
.controller('ExampleController', ['$scope', function ($scope) {
        $scope.user = { name: 'say', data: '' };
        $scope.cancel = function (e) {
        if (e.keyCode == 27) {
        $scope.userForm.userName.$rollbackViewValue();
        }
        };
}]).controller('ExampleController1', ['$scope', function ($scope) {
        $scope.user = { name: 'say' };
}]).controller('ExampleController2', ['$scope', function ($scope) {
        var _name = 'Brian';
        $scope.user = {
        name: function (newName) {
        return angular.isDefined(newName) ? (_name = newName) : _name;
        }
        };
}]);;
        </script>
</head>
<body>
        <div ng-controller="ExampleController">
        <form name="userForm">
            Name:
        <input type="text" name="userName"
        ng-model="user.name"
        ng-model-options="{ updateOn: 'blur' }"
        ng-keyup="cancel($event)" /><br />
            Other data:
        <input type="text" ng-model="user.data" /><br />
        </form>
        <pre>user.name = <span ng-bind="user.name"></span></pre>
        </div>
        <div ng-controller="ExampleController1">
        <form name="userForm">
            Name:
        <input type="text" name="userName"
        ng-model="user.name"
        ng-model-options="{ debounce: 1000 }" />
        <button ng-click="userForm.userName.$rollbackViewValue(); user.name=''">Clear</button><br />
        </form>
        <pre>user.name = <span ng-bind="user.name"></span></pre>
        </div>
        <div ng-controller="ExampleController2">
        <form name="userForm">
            Name:
        <input type="text" name="userName"
        ng-model="user.name"
        ng-model-options="{ getterSetter: true }" />
        </form>
        <pre>user.name = <span ng-bind="user.name()"></span></pre>
        </div>
</body>
</html>
